results of the following outcome variables:

    \item

    \item

    \item

    \end{itemize}
  \item
    Pattern of consumption

    \begin{itemize}
    \tightlist
    \item
      Grams
    \item
      Frequency
    \end{itemize}
  \end{enumerate}
\item
  Impulsivity

  \begin{enumerate}
  \def\labelenumi{\arabic{enumi}.}
  \tightlist
  \item

    \begin{itemize}
    \tightlist
    \item
      Cognitive subscale
    \item
      Motor subscale
    \item
      Non-planning subscale
    \item
      Total score

  \end{enumerate}

  General improvement

  \begin{enumerate}
  \def\labelenumi{\arabic{enumi}.}

  \item
    Subjective improvement report
  \end{enumerate}
\end{itemize}
